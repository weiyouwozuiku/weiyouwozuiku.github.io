### [2319. 判断矩阵是否是一个 X 矩阵](https://leetcode.cn/problems/check-if-matrix-is-x-matrix/)

```cpp
class Solution {
public:
    bool checkXMatrix(vector<vector<int>>& grid) {
        int len=grid.size();
        for(int i=0;i<len;++i){
            for(int j=0;j<len;++j){
                if(i==j||len-j-1==i){
                    if(!grid[i][j]) return false;
                }else{
                    if(grid[i][j]) return false;
                }
            }
        }
        return true;
    }
};
```
